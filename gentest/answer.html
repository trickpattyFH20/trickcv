function findGenIndices(generator, M) {

    let done = false,
        output = [],
        values = [];
    
    while(!done) {
    	let next = generator.next();
        if(!next.done) {
            values.push(next.value);
        } else {
       	    done = true;
        }
    }

    output.push(values[M - 1]);
    output.push(values[(parseInt(values.length) - M)]);

    return output;
}
