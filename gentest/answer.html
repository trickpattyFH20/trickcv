function findGenIndices(gen, n) {

    let done = false,
        output = [],
        values = [];

    while(!done) {
    	let next = gen.next();
        if(!next.done) {
            values.push(next.value);
        } else {
       	    done = true;
        }
    }

    output.push(values[n - 1]);
    output.push(values[(parseInt(values.length) - n)]);

    return output;
}
